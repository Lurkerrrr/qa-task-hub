pm.test("Status code is 200", function () {
    pm.expect(pm.response.code).to.be.equal(200);
});

pm.test("Response sets secure cookie", function () {
    // Secure check: verify the Set-Cookie header contains the token
    pm.expect(pm.response.headers.get('Set-Cookie')).to.contain('token=');

    // Extract token for subsequent non-browser Postman tests
    var cookieString = pm.response.headers.get('Set-Cookie');
    var token = cookieString.split('token=')[1].split(';')[0];
    pm.environment.set("token", token);
});
