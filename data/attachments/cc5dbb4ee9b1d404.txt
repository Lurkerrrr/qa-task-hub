pm.test("Status code is 200", function () {
    pm.expect(pm.response.code).to.be.equal(200);
});

pm.test("Extract Auth and CSRF Tokens", function () {
    var cookies = pm.response.headers.all().filter(h => h.key.toLowerCase() === 'set-cookie');
    cookies.forEach(c => {
        var cookieString = c.value.trim();
        if (cookieString.startsWith('token=')) {
            var token = cookieString.split('token=')[1].split(';')[0];
            pm.environment.set("token", token);
        }
        if (cookieString.startsWith('csrf_token=')) {
            var csrfToken = cookieString.split('csrf_token=')[1].split(';')[0];
            pm.environment.set("csrf_token", csrfToken);
        }
    });
});
